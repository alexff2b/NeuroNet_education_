52 iaVector iaNetwork::ForwardPropogation(iaVector inputVector, activation foncActiv) { // Получение выхода сети (прямое распространение)
53 
54  for (int l = 0; l < this->layersNumber; l++) { // Перебор слоев
55  if (l == 0) { // Если активный слой нулевой
56  for (int i = 0; i < inputVector.sizeOfVector(); i++) { // Перебор нейронов
57  this->layers[l].x[i] = inputVector[i]; // Получить входные сигналы
58  }
59  } else { // иначе если это внутренный или выходной слои
60  for (int i = 0; i < this->layers[l - 1].z.sizeOfVector(); i++) { // Перебор нейронов
61  this->layers[l].x[i] = this->layers[l - 1].z[i]; // Получить входные сигналы от предыдущего слоя
62  }
63  }
64  for (int i = 0; i < this->weights[l].numberOfVectors(); i++) { // Перебор нейронов
65  double y = 0; // Выход функции активации нейрона
66  for (int j = 0; j < this->weights[l].sizeOfVectors(); j++) // Перебор входов в нейрон
67  y += this->weights[l][i][j] * this->layers[l].x[j]; // Подсчет взвешенной суммы входных сигналов
68 
69  switch (foncActiv) {
70  case Hyperbolic_Tangent: {
71  this->layers[l].z[i] = tan(y); // Активация функции гиперболическим тангенсом (-1, 1)
72  this->layers[l].fD[i] = 1 - pow(this->layers[l].z[i], 2); // *this->layers[l].z[i]; // Производная гиперболического тангенса
73  break; }
74  case Arctangent: {
75  this->layers[l].z[i] = atan(y);
76  this->layers[l].fD[i] = 1 / (y * y + 1);
77  break; }
78  case Sigmoid: {
79  this->layers[l].z[i] = 1 / (1 + exp(-y)); // Активация функции сигмойдной функцией (0, 1)
80  this->layers[l].fD[i] = layers[l].z[i] * (-layers[l].z[i]); // Производная сигмойдной функции
81  break; }
82  case ReLu: {
83  this->layers[l].z[i] = y > 0 ? y : 0; // Активация функции ReLU [0, y)
84  this->layers[l].fD[i] = y > 0 ? 1 : 0; // Производная функции ReLU
85  break; }
86  case Gaussian: {
87  this->layers[l].z[i] = exp(-y * y); // Активация функции Гаусса (0, 1]
88  this->layers[l].fD[i] = -2 * y * exp(-y * y); // Производная функции Гаусса
89  break; }
90  case Sin: {
91  this->layers[l].z[i] = sin(y);
92  this->layers[l].fD[i] = cos(y);
93  break; }
94  case Sinc: {
95  this->layers[l].z[i] = (y == 0 ? 1 : sin(y) / y);
96  this->layers[l].fD[i] = (y == 0 ? 0 : cos(y) / y - sin(y) / y);
97  break; }
98  default: { // Identity
99  this->layers[l].z[i] = y;
100  this->layers[l].fD[i] = 1;
101  break; }
102  }
103  }
104  }
105  return layers[layersNumber - 1].z; // Возврат результата
106 }

108 void iaNetwork::BackwardPropogation(iaVector outputVector, double* error) { // Подготовка к обновлению весовых коэффициентов (обратное распространение)
109  *error = 0; // Сброс величины ошибки в ноль
110  int last = this->layersNumber - 1; // Установка предыдущего перед последним слоя
111  for (int i = 0; i < outputVector.sizeOfVector(); i++) { // Перебор нейронов
112  double err = this->layers[last].z[i] - outputVector[i]; // Вычисление ошибки
113  this->deltas[last][i] = err * this->layers[last].fD[i]; // Вычисление дельты
114  *error = err * err / 2; // Подсчет величины ошибки (среднеквадратичное отклонение)
115  }
116  for (int l = last; l > 0; l--) { // Обратный перебор слоев
117  for (int j = 0; j < this->weights[l].sizeOfVectors(); j++) { // Перебо нейронов
118  this->deltas[l - 1][j] = 0; // Сброс дельты
119  for (int i = 0; i < this->weights[l].numberOfVectors(); i++) // Перебор входов в нейрон
120  this->deltas[l - 1][j] += this->weights[l][i][j] * this->deltas[l][i]; // Коррекция дельты
121  this->deltas[l - 1][j] *= this->layers[l - 1].fD[j]; // Коррекция последней дельты
122  }
123  }
124 }

126 void iaNetwork::UpdateWeights(double alpha) { // Обновление весовых коэффициентов (alpha - скорость обучения)
127  for (int l = 0; l < this->layersNumber; l++) { // Перебор слоев
128  for (int i = 0; i < this->weights[l].numberOfVectors(); i++) { // Перебор нейронов
129  for (int j = 0; j < this->weights[l].sizeOfVectors(); j++) { // Перебор входов в нейрон
130  this->weights[l][i][j] -= alpha * this->deltas[l][i] * this->layers[l].x[j]; // Обновление весового коэффициента
131  }
132  }
